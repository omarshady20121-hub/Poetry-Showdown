/********************************************************************
* 2. AUDIO SYNTHESIZER ENGINE (WEB AUDIO API)
********************************************************************/
let audioCtx = null;

function initAudio() {
   if (!audioCtx) {
       audioCtx = new (window.AudioContext || window.webkitAudioContext)();
   }
}

function playSound(type) {
   if (isMuted) return;
   try {
       initAudio();
       const osc = audioCtx.createOscillator();
       const gain = audioCtx.createGain();
       osc.connect(gain);
       gain.connect(audioCtx.destination);

       const now = audioCtx.currentTime;

       if (type === 'correct') {
           osc.type = 'triangle';
           osc.frequency.setValueAtTime(523.25, now); // C5
           osc.frequency.setValueAtTime(659.25, now + 0.1); // E5
           gain.gain.setValueAtTime(0.3, now);
           gain.gain.exponentialRampToValueAtTime(0.01, now + 0.4);
           osc.start(now);
           osc.stop(now + 0.4);
       } else if (type === 'wrong') {
           osc.type = 'sawtooth';
           osc.frequency.setValueAtTime(130.81, now); // C3
           osc.frequency.linearRampToValueAtTime(85, now + 0.3);
           gain.gain.setValueAtTime(0.4, now);
           gain.gain.exponentialRampToValueAtTime(0.01, now + 0.35);
           osc.start(now);
           osc.stop(now + 0.35);
       } else if (type === 'daily_double') {
           osc.type = 'square';
           osc.frequency.setValueAtTime(220, now); // A3
           osc.frequency.setValueAtTime(330, now + 0.1); // E4
           osc.frequency.setValueAtTime(440, now + 0.2); // A4
           osc.frequency.setValueAtTime(660, now + 0.3); // E5
           gain.gain.setValueAtTime(0.2, now);
           gain.gain.exponentialRampToValueAtTime(0.01, now + 0.6);
           osc.start(now);
           osc.stop(now + 0.6);
       } else if (type === 'steal') {
           osc.type = 'sine';
           osc.frequency.setValueAtTime(800, now);
           osc.frequency.linearRampToValueAtTime(300, now + 0.2);
           osc.frequency.linearRampToValueAtTime(900, now + 0.4);
           gain.gain.setValueAtTime(0.25, now);
           gain.gain.exponentialRampToValueAtTime(0.01, now + 0.45);
           osc.start(now);
           osc.stop(now + 0.45);
       } else if (type === 'winner') {
           osc.type = 'sine';
           osc.frequency.setValueAtTime(261.63, now); // C4
           osc.frequency.setValueAtTime(329.63, now + 0.15); // E4
           osc.frequency.setValueAtTime(392.00, now + 0.3); // G4
           osc.frequency.setValueAtTime(523.25, now + 0.45); // C5
           gain.gain.setValueAtTime(0.4, now);
           gain.gain.exponentialRampToValueAtTime(0.02, now + 0.9);
           osc.start(now);
           osc.stop(now + 0.9);
       }
   } catch (e) {
       console.log("Audio contextual playback blocked or unsupported:", e);
   }
}

function toggleMute() {
   isMuted = !isMuted;
   const btn = document.getElementById('mute-toggle-btn');
   if (isMuted) {
       btn.innerHTML = "<span>🔇</span> Sound Muted";
       btn.style.background = "rgba(230, 57, 70, 0.2)";
   } else {
       btn.innerHTML = "<span>🔊</span> Sound On";
       btn.style.background = "var(--bg-card)";
   }
}

function renderJeopardyBoard() {
   const grid = document.getElementById('jeopardy-grid-target');
   grid.innerHTML = '';

   categoriesData.forEach((cat, catIdx) => {
       const col = document.createElement('div');
       col.className = 'category-column';

       const header = document.createElement('div');
       header.className = 'category-header';
       header.innerText = cat.name;
       col.appendChild(header);

       cat.questions.forEach((q, qIdx) => {
           const cell = document.createElement('div');
           cell.className = 'question-cell';
           cell.id = `cell-${catIdx}-${qIdx}`;
           cell.innerText = q.value;
           cell.onclick = () => launchQuestion(catIdx, qIdx);
           col.appendChild(cell);
       });

       grid.appendChild(col);
   });
}

function renderLeaderboard() {
   const target = document.getElementById('leaderboard-target');
   let sortedTeams = [...teams].sort((a, b) => b.score - a.score);
   const highIndex = sortedTeams.length > 0 ? sortedTeams[0].id : null;

   target.innerHTML = '';

   sortedTeams.forEach((team, rank) => {
       const isFirst = (team.id === highIndex && team.score > 0);
       
       const card = document.createElement('div');
       card.className = `team-card ${isFirst ? 'first-place' : ''} ${team.isFrozen ? 'frozen' : ''}`;
       
       card.innerHTML = `
           <div class="team-header">
               <input type="text" class="team-name-input" value="${team.name}" onchange="updateTeamName(${team.id}, this.value)">
               <div class="team-score">${team.score}</div>
           </div>
           <div class="score-actions">
               <button class="score-btn mini-btn" style="background:#22c55e22; border-color:#22c55e;" onclick="adjustScoreDirect(${team.id}, 100)">+100</button>
               <button class="score-btn mini-btn" style="background:#ef444422; border-color:#ef4444;" onclick="adjustScoreDirect(${team.id}, -100)">-100</button>
               <button class="score-btn mini-btn" style="background:#a2d2ff22; border-color:#0077b6;" onclick="toggleFreezeTeam(${team.id})">❄️</button>
           </div>
       `;
       target.appendChild(card);
   });

   populateWagerTeamsDropdown();
}

function updateTeamName(id, val) {
   teams[id].name = val;
   populateWagerTeamsDropdown();
   renderFinalBattleControls();
}

function adjustScoreDirect(id, val) {
   teams[id].score += val;
   renderLeaderboard();
}

function toggleFreezeTeam(id) {
   teams[id].isFrozen = !teams[id].isFrozen;
   renderLeaderboard();
}

function switchView(viewKey) {
   document.querySelectorAll('.view-panel').forEach(p => p.classList.remove('active'));
   document.querySelectorAll('.tab-btn').forEach(b => b.classList.remove('active'));

   if (viewKey === 'board') {
       document.getElementById('panel-board').classList.add('active');
       document.getElementById('tab-board').classList.add('active');
   } else if (viewKey === 'speed') {
       document.getElementById('panel-speed').classList.add('active');
       document.getElementById('tab-speed').classList.add('active');
   } else if (viewKey === 'final') {
       document.getElementById('panel-final').classList.add('active');
       document.getElementById('tab-final').classList.add('active');
   }
}

function launchQuestion(catIdx, qIdx) {
   const cell = document.getElementById(`cell-${catIdx}-${qIdx}`);
   if (cell.classList.contains('completed')) return;

   const qData = categoriesData[catIdx].questions[qIdx];
   currentActiveQuestion = { ...qData, catIdx, qIdx, cellId: `cell-${catIdx}-${qIdx}` };

   document.getElementById('modal-category-label').innerText = categoriesData[catIdx].name;
   document.getElementById('modal-points-label').innerText = `${qData.value} Points`;
   document.getElementById('modal-question-text').innerHTML = qData.q;
   document.getElementById('modal-answer-holder').style.display = 'none';
   document.getElementById('reveal-answer-btn').style.display = 'block';

   const wagerBox = document.getElementById('modal-wager-box');
   if (qData.type === 'daily_double' || qData.type === 'risk_all') {
       wagerBox.style.display = 'block';
       activeWagerAmount = 0;
       playSound('daily_double');
   } else {
       wagerBox.style.display = 'none';
   }

   if (categoriesData[catIdx].name === "Fast Rhymes") {
       setMasterTimer(10);
       startMasterTimer();
   }

   buildModalScoringGrid();
   document.getElementById('main-question-modal').style.display = 'flex';
}

function populateWagerTeamsDropdown() {
   const select = document.getElementById('wager-team-select');
   if (!select) return;
   select.innerHTML = '';
   teams.forEach(t => {
       const opt = document.createElement('option');
       opt.value = t.id;
       opt.innerText = t.name;
       select.appendChild(opt);
   });

   const speedSelect = document.getElementById('speed-team-picker');
   if (speedSelect && speedSelect.children.length === 0) {
       teams.forEach(t => {
           const opt = document.createElement('option');
           opt.value = t.id;
           opt.innerText = t.name;
           speedSelect.appendChild(opt);
       });
   }
}

function lockWagerValue() {
   const teamId = parseInt(document.getElementById('wager-team-select').value);
   const inputVal = parseInt(document.getElementById('wager-input-field').value) || 0;
   const currentScore = teams[teamId].score;

   let maxAllowed = currentScore > 0 ? currentScore : 1000;
   if (currentActiveQuestion.type === 'risk_all') {
       maxAllowed = currentScore;
   }

   if (inputVal > maxAllowed) {
       alert(`Invalid Wager! Maximum permissible point wager configuration is currently ${maxAllowed}`);
       return;
   }

   activeWagerAmount = inputVal;
   activeWagerTeamIndex = teamId;
   alert(`Wager Locked! ${teams[teamId].name} has put forward ${activeWagerAmount} points on this poetic prompt.`);
   document.getElementById('modal-wager-box').style.display = 'none';
}

function revealActiveAnswer() {
   const holder = document.getElementById('modal-answer-holder');
   const txt = document.getElementById('modal-answer-text');
   
   if (currentActiveQuestion.a) {
       txt.innerHTML = currentActiveQuestion.a;
   } else {
       txt.innerHTML = "Special Event execution. Apply scoring manually based on constraints.";
   }

   holder.style.display = 'block';
   document.getElementById('reveal-answer-btn').style.display = 'none';
}

           <p>${t.name}</p>
           <div class="award-btn-group">
               <button class="mini-btn" style="background:#22c55e;" onclick="awardModalPoints(${t.id}, true)">✔</button>
               <button class="mini-btn" style="background:#ef4444;" onclick="awardModalPoints(${t.id}, false)">✖</button>
           </div>
       `;
       target.appendChild(row);
   });
}

function awardModalPoints(teamId, isCorrect) {
   let baseValue = currentActiveQuestion.value;

   if (currentActiveQuestion.type === 'double_points') {
       baseValue = baseValue * 2;
   }

   if (activeWagerAmount > 0 && teamId === activeWagerTeamIndex) {
       baseValue = activeWagerAmount;
   }

   if (isCorrect) {
       teams[teamId].score += baseValue;
       playSound('correct');
   } else {
       teams[teamId].score -= baseValue;
       playSound('wrong');
   }

   renderLeaderboard();
   buildModalScoringGrid();
}

function closeQuestionModal() {
   if (currentActiveQuestion && currentActiveQuestion.cellId) {
       const cell = document.getElementById(currentActiveQuestion.cellId);
       if (cell) {
           cell.classList.add('completed');
           cell.onclick = null;
       }
   }

   document.getElementById('main-question-modal').style.display = 'none';
   stopMasterTimer();
   currentActiveQuestion = null;
   activeWagerAmount = 0;
   activeWagerTeamIndex = null;

   checkGameCompletionCondition();
}

   }

}

function submitSpeedAnswer(qIdx, selectedIdx) {
   const chosenTeamId = parseInt(document.getElementById('speed-team-picker').value);
   const correctIdx = speedRoundQuestions[qIdx].correct;

   if (selectedIdx === correctIdx) {
       teams[chosenTeamId].score += 200;
       playSound('correct');
       alert("Correct! 200 points awarded.");
   } else {
       playSound('wrong');
       alert(`Incorrect. The proper solution was: ${speedRoundQuestions[qIdx].options[correctIdx]}`);
   }

   renderLeaderboard();

   const currentCard = document.getElementById(`speed-q-${qIdx}`);
   currentCard.classList.remove('active');

   const nextCard = document.getElementById(`speed-q-${qIdx + 1}`);
   if (nextCard) {
       nextCard.classList.add('active');
   } else {
       alert("The Speed Round challenge matrix is fully exhausted! Check the live leaderboard.");
       const finishNotice = document.createElement('div');
       finishNotice.innerHTML = `<h3 style='color:var(--success); margin-top:20px;'>⚡ Speed Round Complete!</h3>`;
       document.getElementById('speed-questions-container').appendChild(finishNotice);
   }
}

/********************************************************************
* 11. HIGH-PERFORMANCE CANVAS-BASED CONFETTI SYSTEM
********************************************************************/
let confettiInterval = null;
const canvas = document.getElementById("confetti-canvas");
const ctx = canvas.getContext("2d");
let confettiPieces = [];

function initConfetti() {
   canvas.width = window.innerWidth;
   canvas.height = window.innerHeight;
   confettiPieces = [];
   for (let i = 0; i < 150; i++) {
       confettiPieces.push({
           x: Math.random() * canvas.width,
           y: Math.random() * canvas.height - canvas.height,
           r: Math.random() * 6 + 4,
           d: Math.random() * canvas.height,
           color: `hsl(${Math.random() * 360}, 100%, 60%)`,
           tilt: Math.random() * 10 - 5,
           tiltAngleIncremental: Math.random() * 0.07 + 0.02,
           tiltAngle: 0
       });
   }
   if (confettiInterval) clearInterval(confettiInterval);
   confettiInterval = setInterval(drawConfettiLoop, 20);
}

function drawConfettiLoop() {
   ctx.clearRect(0, 0, canvas.width, canvas.height);
   confettiPieces.forEach((p, idx) => {
       p.tiltAngle += p.tiltAngleIncremental;
       p.y += (Math.cos(p.d) + 3 + p.r / 2) / 2;
       p.x += Math.sin(p.tiltAngle);
       p.tilt = Math.sin(p.tiltAngle - idx / 3) * 15;

       if (p.y > canvas.height) {
           confettiPieces[idx] = {
               x: Math.random() * canvas.width,
               y: -20,
               r: p.r,
               d: p.d,
               color: p.color,
               tilt: p.tilt,
               tiltAngleIncremental: p.tiltAngleIncremental,
               tiltAngle: p.tiltAngle
           };
       }

       ctx.beginPath();
       ctx.lineWidth = p.r;
       ctx.strokeStyle = p.color;
       ctx.moveTo(p.x + p.tilt + p.r / 2, p.y);
       ctx.lineTo(p.x + p.tilt, p.y + p.tilt + p.r / 2);
       ctx.stroke();
   });
}

function stopConfetti() {
   if (confettiInterval) {
       clearInterval(confettiInterval);
       confettiInterval = null;
   }
   ctx.clearRect(0, 0, canvas.width, canvas.height);
}

window.addEventListener('resize', () => {
   canvas.width = window.innerWidth;
   canvas.height = window.innerHeight;
});
